 99     public static String draw(String[] newDeck, int cardIndex)
100     {
101         // Store a suit to a String varible based upon index
102         
103         String suit;
104         switch (cardIndex)
105         {
106             case 0:
107             case 10:

110                 suit = "A";

112             case 40:

import java.io.*;
import java.util.*;
public class HolmesJHW3 {

    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) throws FileNotFoundException
    {
        File studentList = new File("studentList.txt");
        PrintWriter fileWriter = new PrintWriter("studentList.txt");
        Random rand = new Random();
        Scanner scan = new Scanner(System.in);
        Scanner fileInput = new Scanner(studentList);

        // Create an Array for class levels
        String[] classLevels = new String[4];
        classLevels[0] = "freshman";
        classLevels[1] = "sophomore";
        classLevels[2] = "junior";
        classLevels[3] = "senior";

        // Create new Array for 1000 student records
        String[] allStudents = new String[1000];

        // Store 1000 random student records to txt file
        for (int i = 0; i < allStudents.length; i++)
        {
            fileWriter.print("FirstName" + (i + 1) + " ");
            fileWriter.print("LastName" + (i + 1) + " ");
            fileWriter.print(classLevels[rand.nextInt(4)] + " ");
            double gpa = (rand.nextInt(201) + 200) / 100.00;
            fileWriter.printf("%4.2f \n", gpa);
        }
        fileWriter.close();

        // Copy student records from txt file to the Array for student records
        int j = 0;
        while (fileInput.hasNext())
        {
            String student = fileInput.nextLine();
            allStudents[j] = student;
            j++;
        }

        // Create an new ArrayList and fill it with students requested for report
        ArrayList<String> reportStudents = new ArrayList<>();
        System.out.print("Enter a student last name, 'quit' when finished: ");
        String lName = scan.nextLine() + " ";
        while (!lName.contains("quit"))
        {
            reportStudents.add(lName);
            System.out.print("Enter a student last name, 'quit' when finished: ");
            lName = scan.nextLine() + " ";
        }

        // Print each Student record for each requested name
        System.out.println("\nStudent Report:");
        for (String record : allStudents)
        {
            for (String students : reportStudents)
                {
                    if (record.contains(students))
                    {
                        System.out.println(record);
                    }
                }


        fileInput.close();
        }

    }
}
